const sampleBlogs = [
  { id: '1', title: 'Getting started with React', content: 'React is a powerful library for building UIs. This post introduces components, state, and props.', tags: ['react', 'frontend'], date: new Date().toLocaleString() },
  { id: '2', title: 'How to use LocalStorage', content: 'LocalStorage is a simple way to persist small amounts of data on the client.', tags: ['webdev', 'localstorage'], date: new Date().toLocaleString() },
];

export default function App() {
  const [blogs, setBlogs] = useState(() => {
    try { const raw = localStorage.getItem('blogs'); if (!raw) return sampleBlogs; const parsed = JSON.parse(raw); return Array.isArray(parsed) && parsed.length ? parsed : sampleBlogs; } catch { return sampleBlogs; }
  });

  useEffect(() => { if (!document.getElementById('app-styles')) { const s = document.createElement('style'); s.id = 'app-styles'; s.innerHTML = styles; document.head.appendChild(s); } }, []);
  useEffect(() => { localStorage.setItem('blogs', JSON.stringify(blogs)); }, [blogs]);

  const addBlog = (newBlog) => setBlogs(prev => [...prev, newBlog]);
  const deleteBlog = (id) => setBlogs(prev => prev.filter(b => b.id !== id));

  return (
    <Router>
      <div className="app-container">
        <Sidebar />
        <main className="main-content">
          <Routes>
            <Route path="/" element={<BlogList blogs={blogs} />} />
            <Route path="/create" element={<BlogForm onSubmit={addBlog} />} />
            <Route path="/blog/:id" element={<BlogDetail blogs={blogs} onDelete={deleteBlog} />} />
          </Routes>
        </main>
      </div>
    </Router>
  );
}

function Sidebar() {
  const [searchTag, setSearchTag] = useState('');
  const navigate = useNavigate();
  const handleSearch = (e) => { e.preventDefault(); const tag = searchTag.trim(); if(tag) navigate(`/?tag=${encodeURIComponent(tag)}`); else navigate('/'); };
  return (
    <aside className="sidebar">
      <h2>📘 MyBlog</h2>
      <nav>
        <Link to="/" className="nav-btn">🏠 Home</Link>
        <Link to="/create" className="nav-btn">✍️ Create Blog</Link>
      </nav>
      <form onSubmit={handleSearch} className="search-box">
        <input type="text" placeholder="#Search hashtag" value={searchTag} onChange={(e)=>setSearchTag(e.target.value)} />
        <button type="submit">🔍</button>
      </form>
    </aside>
  );
}

function BlogForm({ onSubmit }) {
  const [title,setTitle]=useState('');
  const [content,setContent]=useState('');
  const [tags,setTags]=useState('');
  const navigate = useNavigate();

  const handleSubmit = (e) => {
    e.preventDefault();
    const tagArray = tags.split(',').map(t=>t.trim().toLowerCase()).filter(Boolean);
    if(!title.trim() || !content.trim()){ alert('Please fill title and content'); return; }
    if(tagArray.length<2){ alert('Enter at least 2 hashtags'); return; }
    const newBlog={id:Date.now().toString(),title:title.trim(),content:content.trim(),tags:tagArray,date:new Date().toLocaleString()};
    onSubmit(newBlog);
    setTitle(''); setContent(''); setTags('');
    navigate('/');
  };

  return (
    <div className="form-container">
      <h2>Write a New Blog</h2>
      <form onSubmit={handleSubmit}>
        <input placeholder="Blog Title" value={title} onChange={(e)=>setTitle(e.target.value)} />
        <textarea placeholder="Write your blog content..." rows={8} value={content} onChange={(e)=>setContent(e.target.value)} />
        <input placeholder="Comma separated hashtags" value={tags} onChange={(e)=>setTags(e.target.value)} />
        <button type="submit">Publish</button>
      </form>
    </div>
  );
}

function BlogList({ blogs }) {
  const location=useLocation();
  const params=new URLSearchParams(location.search);
  const filterTag=params.get('tag');
  const filtered=filterTag ? blogs.filter(b=>(b.tags||[]).includes(filterTag.toLowerCase())) : blogs;
  if(!filtered.length) return <p className="empty">No blogs found. Try another tag!</p>;
  return (
    <div className="blog-list">
      {filtered.map(blog=>(<Link key={blog.id} to={`/blog/${blog.id}`} className="no-decoration"><BlogCard blog={blog} /></Link>))}
    </div>
  );
}


function BlogCard({ blog }) {
  return (
    <div className="blog-card">
      <h3>{blog.title}</h3>
      <p className="date">{blog.date}</p>
      <p>{(blog.content||'').slice(0,200)}{(blog.content||'').length>200?'...':''}</p>
      <div className="tags">{(blog.tags||[]).map((t,i)=><span key={i} className="tag">#{t}</span>)}</div>
    </div>
  );
}

function BlogDetail({ blogs, onDelete }) {
  const { id } = useParams();
  const navigate = useNavigate();
  const blog = (blogs||[]).find(b=>b.id===id);
  if(!blog) return <p className="empty">Blog not found.</p>;

  const handleTagClick=(tag)=>{ navigate(`/?tag=${encodeURIComponent(tag)}`); };
  const handleDelete=()=>{ if(window.confirm('Delete this blog?')){ onDelete(blog.id); navigate('/'); } };
  const related=(blogs||[]).filter(b=>b.id!==blog.id&&(b.tags||[]).some(t=>(blog.tags||[]).includes(t)));

  return (
    <div className="blog-detail">
      <Link to="/" className="back-link">⬅ Back</Link>
      <h2>{blog.title}</h2>
      <p className="date">{blog.date}</p>
      <p style={{whiteSpace:'pre-wrap'}}>{blog.content}</p>
      <div className="tags" style={{marginTop:12}}>{(blog.tags||[]).map((t,i)=><span key={i} className="tag" onClick={()=>handleTagClick(t)}>#{t}</span>)}</div>
      <button className="delete-btn" onClick={handleDelete}>🗑️ Delete</button>
      {related && related.length>0 && (<div style={{marginTop:24}}><h4>Related posts</h4><div className="blog-list">{related.slice(0,3).map(r=><Link key={r.id} to={`/blog/${r.id}`} className="no-decoration"><BlogCard blog={r} /></Link>)}</div></div>)}
    </div>
  );
}
